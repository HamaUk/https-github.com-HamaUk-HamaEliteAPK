    databaseURL: "https://hama-elite-sync-default-rtdb.firebaseio.com",
    projectId: "hama-elite-sync",

    appId: "1:306546806510:web:38dffe0bc504ca26f92dff"
  };

  const app = initializeApp(firebaseConfig);
  const db = getDatabase(app);
  const auth = getAuth(app);

  let currentMethod = 'xtream';
  let addMethod = 'xtream';
  let editingCode = null;
  let refreshTimer = null;

  window.setMethod = (method) => {
    currentMethod = method;
    document.getElementById('tab-xtream').className = 'tab ' + (method === 'xtream' ? 'active' : '');
    document.getElementById('tab-m3u').className = 'tab ' + (method === 'm3u' ? 'active' : '');

    document.getElementById('fields-file').style.display = (method === 'file' ? 'block' : 'none');
  };

  window.setAddMethod = (method) => {
    addMethod = method;
    document.getElementById('add-tab-xtream').className = 'tab ' + (method === 'xtream' ? 'active' : '');
    document.getElementById('add-tab-m3u').className = 'tab ' + (method === 'm3u' ? 'active' : '');
    document.getElementById('add-tab-file').className = 'tab ' + (method === 'file' ? 'active' : '');
    const fields = document.getElementById('add-fields');
    if (method === 'xtream') {
      fields.innerHTML = `<div class="form-group"><label>Server</label><input type="text" id="add-server"></div>
      <div class="row">
        <div class="form-group"><label>User</label><input type="text" id="add-user"></div>
        <div class="form-group"><label>Pass</label><input type="text" id="add-pass"></div>
      </div>`;
    } else if (method === 'm3u') {
      fields.innerHTML = `<div class="form-group"><label>M3U URL</label><input type="text" id="add-url"></div>`;
    } else {
      fields.innerHTML = `
        <div class="form-group"><label>Upload M3U File</label>
          <div class="file-upload-wrapper">
            <label for="add-file" class="file-upload-label" style="padding:15px">
              <span>CHOOSE M3U FILE</span>
              <div id="addFileName" class="file-selected-name">No file selected</div>
            </label>
            <input type="file" id="add-file" class="file-upload-input" accept=".m3u,.m3u8" onchange="updateFileName(this, 'addFileName')">
          </div>
        </div>
      `;
    }
  };

  const readFileContent = (file) => {
    return new Promise((resolve, reject) => {
      const reader = new FileReader();
      reader.onload = (e) => resolve(e.target.result);
      reader.onerror = (e) => reject(new Error("Failed to read file"));
      reader.readAsText(file);
    });
  };

  window.updateFileName = (input, displayId) => {
    const file = input.files[0];
    document.getElementById(displayId).innerText = file ? file.name : "No file selected";
  };

  document.getElementById('syncBtn').onclick = async () => {
    const code = document.getElementById('linkCode').value;
    const status = document.getElementById('status');
    const btn = document.getElementById('syncBtn');

    if (!code || code.length < 6) {
      status.innerText = "Please enter a valid 6-digit code";
      status.className = "status-msg error";
      return;
    }

    btn.disabled = true;
    btn.innerText = "SENDING TO CLOUD...";
    status.innerText = "";

    try {
      const dbRef = ref(db, 'sync/' + code);
      let payload = { method: currentMethod, timestamp: Date.now() };

      if (currentMethod === 'xtream') {
        payload.server = document.getElementById('server').value;
        payload.user = document.getElementById('user').value;
        payload.pass = document.getElementById('pass').value;
      } else if (currentMethod === 'm3u') {
        payload.url = document.getElementById('m3uUrl').value;
      } else if (currentMethod === 'file') {
        const file = document.getElementById('m3uFile').files[0];
        if (!file) throw new Error("Please select an M3U file");
        payload.method = 'm3u_file';
        payload.content = await readFileContent(file);
        payload.filename = file.name;
      }

      await set(dbRef, payload);

      status.innerText = "✅ DATA SENT! REFRESH YOUR TV APP NOW";
      status.className = "status-msg success";
      btn.innerText = "SEND AGAIN";
      btn.disabled = false;
    } catch (e) {
      status.innerText = "ERROR: " + e.message;
      status.className = "status-msg error";
      btn.disabled = false;
      btn.innerText = "RETRY";
    }
  };

  /* ADMIN LOGIC */
  window.showAdminLogin = () => document.getElementById('adminOverlay').style.display = 'flex';
  window.hideAdminLogin = () => document.getElementById('adminOverlay').style.display = 'none';
  
  window.checkAdminLogin = async () => {
    const email = document.getElementById('adminUser').value;
    const pass = document.getElementById('adminPass').value;
    try {
      await signInWithEmailAndPassword(auth, email, pass);
      hideAdminLogin();
      showAdminDashboard();
    } catch (error) {
      alert("Invalid Credentials. Please try again.");
      document.getElementById('adminPass').value = '';
    }

  window.showAdminDashboard = async () => {
    document.querySelector('.portal-container').style.display = 'none';
    document.getElementById('adminDashboard').style.display = 'block';
    loadUsers();
    refreshTimer = setInterval(() => { if(document.getElementById('autoRefresh').checked) loadUsers(); }, 30000);
  };

  window.hideAdminDashboard = () => {
    signOut(auth).catch(e => console.error("Sign out error", e));
    document.querySelector('.portal-container').style.display = 'block';
    document.getElementById('adminDashboard').style.display = 'none';
    if(refreshTimer) clearInterval(refreshTimer);
  };

  window.loadUsers = async () => {
    const list = document.getElementById('userList');
    try {
      const dbRef = ref(db, 'sync');
      const snapshot = await get(dbRef);
      list.innerHTML = '';
      
      if (snapshot.exists()) {
        const data = snapshot.val();
        window.allUsers = data;
        
        let stats = { total: 0, xtream: 0, m3u: 0 };
        
        Object.keys(data).sort((a,b) => (data[b].timestamp || 0) - (data[a].timestamp || 0)).forEach(code => {
          const u = data[code];
          stats.total++;
          if(u.method === 'xtream') stats.xtream++; else stats.m3u++;

          const tr = document.createElement('tr');
          tr.setAttribute('data-code', code);
          const timeStr = formatTimeAgo(u.timestamp);
          let info = '';
          if (u.method === 'xtream') {
            info = `URL: ${u.server}<br>
                    <span style="font-size:0.8rem">U: ${u.user} | P: <span class="pass-hidden">${u.pass}</span></span>
                    <span class="copy-icon" onclick="copyData('${u.user}:${u.pass}')">📋</span>`;
          } else if (u.method === 'm3u') {
            info = `<span style="font-size:0.8rem">${u.url}</span> <span class="copy-icon" onclick="copyData('${u.url}')">📋</span>`;
          } else {
            info = `<b>FILE:</b> ${u.filename || 'Upload'}`;
          }
          
          tr.innerHTML = `
            <td class="checkbox-col"><input type="checkbox" class="user-check" data-code="${code}"></td>
            <td style="font-weight:900; color:var(--accent)">${code}</td>
            <td><span class="badge ${u.method === 'm3u_file' ? 'file' : u.method}">${u.method.toUpperCase()}</span></td>
            <td style="max-width:350px; overflow:hidden;">${info}</td>
            <td style="font-size:0.75rem">${timeStr}</td>
            <td>
              <button class="action-btn btn-edit" onclick="openEdit('${code}')">EDIT</button>
              <button class="action-btn btn-delete" onclick="deleteAccount('${code}')">DEL</button>
            </td>
          `;
          list.appendChild(tr);
        });
        
        document.getElementById('stat-total').innerText = stats.total;
        document.getElementById('stat-xtream').innerText = stats.xtream;
        document.getElementById('stat-m3u').innerText = stats.m3u;
      } else {
        list.innerHTML = '<tr><td colspan="6" style="text-align:center">No active sync codes found.</td></tr>';
      }
    } catch (e) {
      list.innerHTML = '<tr><td colspan="6" style="text-align:center; color:red">Error: ' + e.message + '</td></tr>';
    }
  };

  const formatTimeAgo = (timestamp) => {
    if (!timestamp) return 'Unknown';
    const seconds = Math.floor((Date.now() - timestamp) / 1000);
    if (seconds < 60) return 'Just now';
    if (seconds < 3600) return Math.floor(seconds/60) + 'm ago';
    if (seconds < 86400) return Math.floor(seconds/3600) + 'h ago';
    return Math.floor(seconds/86400) + 'd ago';
  };

  window.copyData = (text) => {
    navigator.clipboard.writeText(text);
    alert('Copied to clipboard');

  window.bulkDelete = async () => {
    const selected = Array.from(document.querySelectorAll('.user-check:checked')).map(c => c.getAttribute('data-code'));
    if (selected.length === 0) return alert('Select users first');
    if (!confirm(`Delete ${selected.length} selected accounts?`)) return;
    for (const code of selected) {
      await remove(ref(db, 'sync/' + code));
    }
    loadUsers();
  };

  window.deleteAccount = async (code) => {
    if (!confirm(`Are you sure you want to delete the sync data for code ${code}?`)) return;
    try {
      await remove(ref(db, 'sync/' + code));
      loadUsers();
    } catch (e) {
      alert('Error deleting account: ' + e.message);
    }
  };

  window.filterUsers = () => {
    const filter = document.getElementById('adminSearch').value.toUpperCase();
    const rows = document.getElementById('userList').getElementsByTagName('tr');
    for (let i = 0; i < rows.length; i++) {
      if (rows[i].cells.length === 1) continue; // Skip the 'No active sync codes' message
      const textData = rows[i].textContent || rows[i].innerText;
      rows[i].style.display = textData.toUpperCase().indexOf(filter) > -1 ? '' : 'none';
    }
  };

  window.openAddManual = () => {
    document.getElementById('add-code').value = Math.floor(100000 + Math.random() * 900000);
    document.getElementById('addModal').style.display = 'flex';
  };
  window.closeAddModal = () => document.getElementById('addModal').style.display = 'none';

  window.saveAddManual = async () => {
    const code = document.getElementById('add-code').value;
    if (code.length < 6) return alert('Code must be 6 digits');
    
    let payload = { method: addMethod, timestamp: Date.now() };
    if (addMethod === 'xtream') {
      payload.server = document.getElementById('add-server').value;
      payload.user = document.getElementById('add-user').value;
      payload.pass = document.getElementById('add-pass').value;
    } else if (addMethod === 'm3u') {
      payload.url = document.getElementById('add-url').value;
    } else if (addMethod === 'file') {
      const file = document.getElementById('add-file').files[0];
      if (!file) return alert('Select a file first');
      payload.method = 'm3u_file';
      payload.content = await readFileContent(file);
      payload.filename = file.name;
    }
    
    await set(ref(db, 'sync/' + code), payload);
    closeAddModal();
    loadUsers();
  };

  window.exportData = () => {
    if (!window.allUsers) return;
    let csv = "Code,Method,Info,Time\n";
    Object.keys(window.allUsers).forEach(code => {
      const u = window.allUsers[code];
      const info = u.method === 'xtream' ? `${u.server}` : u.url || 'FILE';
      csv += `${code},${u.method},"${info}",${new Date(u.timestamp).toLocaleString()}\n`;
    });
    const blob = new Blob([csv], { type: 'text/csv' });
    const url = window.URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = `hama_elite_sync_${Date.now()}.csv`;
    a.click();
  };

        <div class="form-group"><label>Pass</label><input type="text" id="edit-pass" value="${u.pass}"></div>

      fields.innerHTML = `

      fields.innerHTML = `

            <label for="edit-file" class="file-upload-label" style="padding:15px">

  window.saveEdit = async () => {
    try {
      const dbRef = ref(db, 'sync/' + editingCode);
      const snap = await get(dbRef);
      const u = snap.val();
      
      let payload = { method: u.method, timestamp: Date.now() };
      if (u.method === 'xtream') {
        payload.server = document.getElementById('edit-server').value;
        payload.user = document.getElementById('edit-user').value;
        payload.pass = document.getElementById('edit-pass').value;
      } else if (u.method === 'm3u') {
        payload.url = document.getElementById('edit-url').value;
      } else if (u.method === 'm3u_file') {
        const file = document.getElementById('edit-file').files[0];
        if (file) {
          payload.content = await readFileContent(file);
          payload.filename = file.name;
        } else {
          payload.content = u.content;
          payload.filename = u.filename;
        }
      }
      
      await set(dbRef, payload);
      closeModal();
      loadUsers();
    } catch (e) { alert(e.message); }
  };
